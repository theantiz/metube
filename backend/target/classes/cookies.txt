private String getCookiesFilePath() throws IOException {
    Path cookies = Paths.get("/tmp/cookies.txt");

    if (!Files.exists(cookies)) {
        try (InputStream in = getClass().getClassLoader().getResourceAsStream("cookies.txt")) {
            if (in == null) {
                throw new IOException("cookies.txt missing in resources folder");
            }
            Files.copy(in, cookies);
        }
    }

    return cookies.toAbsolutePath().toString();
}
